    <script>
        // Mobile menu toggle
        const menuBtn = document.getElementById('menuBtn');
        const mobileMenu = document.getElementById('mobileMenu');
        
        menuBtn.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        // Party mode
        const partyModeBtn = document.getElementById('partyModeBtn');
        let partyMode = false;
        let canvas = document.getElementById('partyCanvas');
        let ctx = canvas.getContext('2d');
        
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        
        window.addEventListener('resize', () => {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        });
        
        let particles = [];
        const colors = ['#FF00C3', '#00FFC3', '#00FFF9', '#FBFF00', '#9D00FF'];
        
        class Particle {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.size = Math.random() * 3 + 1;
                this.color = colors[Math.floor(Math.random() * colors.length)];
                this.speedX = Math.random() * 3 - 1.5;
                this.speedY = Math.random() * 3 - 1.5;
                this.opacity = Math.random() * 0.6 + 0.2;
            }
            
            update() {
                this.x += this.speedX;
                this.y += this.speedY;
                
                if (this.x < 0 || this.x > canvas.width) {
                    this.speedX = -this.speedX;
                }
                
                if (this.y < 0 || this.y > canvas.height) {
                    this.speedY = -this.speedY;
                }
            }
            
            draw() {
                ctx.fillStyle = this.color;
                ctx.globalAlpha = this.opacity;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
                ctx.globalAlpha = 0.3;
            }
        }
        
        function initParticles() {
            particles = [];
            for (let i = 0; i < 100; i++) {
                particles.push(new Particle());
            }
        }
        
        function animateParticles() {
            if (!partyMode) return;
            
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            for (let i = 0; i < particles.length; i++) {
                particles[i].update();
                particles[i].draw();
                
                for (let j = i; j < particles.length; j++) {
                    const dx = particles[i].x - particles[j].x;
                    const dy = particles[i].y - particles[j].y;
                    const distance = Math.sqrt(dx * dx + dy * dy);
                    
                    if (distance < 100) {
                        ctx.beginPath();
                        ctx.strokeStyle = particles[i].color;
                        ctx.globalAlpha = 0.2 * (1 - distance / 100);
                        ctx.lineWidth = 0.5;
                        ctx.moveTo(particles[i].x, particles[i].y);
                        ctx.lineTo(particles[j].x, particles[j].y);
                        ctx.stroke();
                        ctx.globalAlpha = 0.3;
                    }
                }
            }
            
            requestAnimationFrame(animateParticles);
        }
        
        partyModeBtn.addEventListener('click', () => {
            partyMode = !partyMode;
            
            if (partyMode) {
                initParticles();
                animateParticles();
                document.body.style.background = 'linear-gradient(rgba(10,10,10,0.9), rgba(10,10,10,0.9))';
                canvas.style.opacity = '0.5';
                partyModeBtn.innerHTML = '<i class="fas fa-times text-2xl"></i>';
                partyModeBtn.classList.remove('from-pink-500', 'to-purple-600');
                partyModeBtn.classList.add('from-purple-600', 'to-pink-500');
            } else {
                document.body.style.background = '#0a0a0a';
                canvas.style.opacity = '0.3';
                partyModeBtn.innerHTML = '<i class="fas fa-glass-cheers text-2xl"></i>';
                partyModeBtn.classList.remove('from-purple-600', 'to-pink-500');
                partyModeBtn.classList.add('from-pink-500', 'to-purple-600');
            }
        });
        
        // Simulate terminal typing
        const terminalLines = [
            "> Connecting to Algorand node...",
            "√ Established secure connection",
            "> Syncing latest blocks...",
            "√ Sync complete (3.2s)",
            "> Analyzing transactions...",
            "√ 128 new transactions in block #24832415",
            "> Checking consensus participation...",
            "√ 83.7% participation rate",
            "> Calculating annual rewards...",
            "√ Estimated 6.4-7.2% APY",
            "> Monitoring network health...",
            "√ All systems optimal",
            "> Preparing real-time updates...",
            "√ Ready for your commands"
        ];
        
        const terminal = document.querySelector('.terminal-text');
        let currentLine = 0;
        
        function typeTerminal() {
            if (currentLine < terminalLines.length) {
                const line = terminalLines[currentLine];
                const p = document.createElement('div');
                p.className = line.startsWith('√') ? 'text-green-400' : '';
                
                if (line.startsWith('>')) {
                    terminal.querySelector('.blink')?.remove();
                }
                
                terminal.appendChild(p);
                
                let i = 0;
                const typing = setInterval(() => {
                    p.textContent = line.substring(0, i);
                    i++;
                    
                    if (i > line.length) {
                        clearInterval(typing);
                        currentLine++;
                        
                        if (line.startsWith('>')) {
                            const blink = document.createElement('span');
                            blink.className = 'blink ml-1';
                            blink.textContent = '_';
                            terminal.appendChild(blink);
                        }
                        
                        setTimeout(typeTerminal, Math.random() * 1000 + 500);
                    }
                }, 20);
            }
        }
        
        // Start typing after a delay
        setTimeout(typeTerminal, 1000);
        
        // Scroll animation for sections
        const sections = document.querySelectorAll('section');
        window.addEventListener('scroll', () => {
            sections.forEach(section => {
                const sectionTop = section.offsetTop;
                const sectionHeight = section.clientHeight;
                const scrollPosition = window.scrollY;
                
                if (scrollPosition > sectionTop - window.innerHeight + sectionHeight * 0.3) {
                    section.style.opacity = '1';
                    section.style.transform = 'translateY(0)';
                }
            });
        });
        
        // Apply initial styles for scroll animation
        sections.forEach((section, index) => {
            section.style.transition = `opacity 0.6s ease ${index * 0.1}s, transform 0.6s ease ${index * 0.1}s`;
            section.style.opacity = '0';
            section.style.transform = 'translateY(20px)';
        });
        
        // Trigger the initial scroll check on load
        window.dispatchEvent(new Event('scroll'));
    </script>
